import "std/net/tcp.zc"

fn main() {
    "Echo Server слушает на :8080";
    let listener = TcpListener::bind("127.0.0.1", 8080).unwrap();

    loop {
        // Принимать новые соединения
        let stream = listener.accept().unwrap();
        let buf: char[1024];
        
        while true {
            let n = stream.read(&buf[0], 1024).unwrap();
            if n == 0 { break; }
            stream.write(&buf[0], n);
        }
    }
}

struct Features {
    transparent_compilation: bool;
    native_c_abi:            bool;
    resource_management:     bool;
    monomorphization:        bool;
    pattern_matching:        bool;
    async_await:             bool;
}

impl Features {
    // Конструктор. Все функции включены!
    fn new() -> Self {
        return Features{
            transparent_compilation: true,
            native_c_abi:            true,
            resource_management:     true,
            monomorphization:        true,
            pattern_matching:        true,
            async_await:             true,
        };
    }

    // Описание языка в одну строку
    fn describe(self) {
        println "Эргономика. Ноль расходов. Чистый C.";
    }
}

fn main() {
    let zc = Features::new();
    zc.describe();
}
