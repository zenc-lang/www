import "std/net/tcp.zc"

fn main() {
    "Echo Server escuchando en :8080";
    let listener = TcpListener::bind("127.0.0.1", 8080).unwrap();

    loop {
        // Aceptar nuevas conexiones
        let stream = listener.accept().unwrap();
        let buf: char[1024];
        
        while true {
            let n = stream.read(&buf[0], 1024).unwrap();
            if n == 0 { break; }
            stream.write(&buf[0], n);
        }
    }
}

struct Features {
    transparent_compilation: bool;
    native_c_abi:            bool;
    resource_management:     bool;
    monomorphization:        bool;
    pattern_matching:        bool;
    async_await:             bool;
}

impl Features {
    // Constructor. ¡Todas las funciones vienen activadas!
    fn new() -> Self {
        return Features{
            transparent_compilation: true,
            native_c_abi:            true,
            resource_management:     true,
            monomorphization:        true,
            pattern_matching:        true,
            async_await:             true,
        };
    }

    // Describir el lenguaje en una línea
    fn describe(self) {
        println "Ergonomía Moderna. Cero Overhead. C Puro.";
    }
}

fn main() {
    let zc = Features::new();
    zc.describe();
}
transparent_compilation

Transpila directamente a GNU C11 limpio y legible: C puro sin cajas negras.

native_c_abi

Llama a cualquier función C, usa cualquier struct C. Cero boilerplate FFI.

resource_management

El trait Drop implícito evita fugas de memoria sin un GC.

monomorphization

Genéricos de cero sobrecarga: especializados en tiempo de compilación, no en ejecución.

async_await

E/S asíncrona de alto rendimiento sin infierno de callbacks ni runtimes ocultos.

pattern_matching

match exhaustivo sobre enums y tuplas para un flujo de control a prueba de balas.

zero_cost_generics

Tipos paramétricos que desaparecen en tiempo de compilación: sin despacho virtual.

Arquitectura Políglota

Zen C se adapta a tu ecosistema. Enlázalo sin problemas con bases de código existentes.

CUDA

--cuda

Escribe kernels de GPU con sintaxis Zen usando atributos launch y @global. 

@global
fn add(a: float*, b: float*) {
    let i = thread_id();
    a[i] += b[i];
}

C++

--cpp

Genera código compatible con C++ para enlazar con librerías. Instancia plantillas en bloques raw. 

raw {
    std::vector<int> get_vec() {
        return {1, 2, 3};
    }
}

Objective-C

--objc

Soporte nativo para macOS. Mezcla la sintaxis [Object message] dentro de tu código de sistemas.

raw {
    id pool = [[NSPool alloc] init];
    [pool drain];
}
